Simple API

from voice_soundboard import VoiceEngine, play_audio

engine = VoiceEngine()
result = engine.speak(
    "Hello! How can I help you today?",
    style="warmly and cheerfully"
)
play_audio(result.audio_path)
# Natural non-speech sounds
result = engine.speak(
    "That's hilarious! [laugh] Oh man, [sigh] I needed that."
)

# Emotion blending
from voice_soundboard import blend_emotions
blend = blend_emotions([("happy", 0.7), ("surprised", 0.3)])
# Result: pleasant_surprise
# Multi-speaker dialogue from script
script = """
[host] Welcome to the show!
[guest] (excited) Thanks for having me!
[host] Let's dive right in.
"""

result = await engine.speak_dialogue(
    script,
    voices={"host": "bm_george", "guest": "af_bella"}
)
play_audio(result.audio_path)
# Clone a voice from 5 seconds of audio
from voice_soundboard import clone_voice

my_voice = clone_voice(
    "sample.wav",
    consent_given=True
)

# Speak in any language with your voice
engine.speak("Bonjour!", voice=my_voice, language="fr")
engine.speak("Guten Tag!", voice=my_voice, language="de")
